FUNCTION calcAVG :
    FOR i:=LOWER_BOUND(Results,1) TO UPPER_BOUND(Results,1) DO
    	IF Results[i] = LTIME#0S THEN
    		EXIT;
    	END_IF

END_FUNCTION
